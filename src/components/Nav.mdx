        {link: '/services', name: 'Services'},
        {link: '/about', name: 'About'},
        # {link: '/gallery', name: 'Gallery'},
        {link: '/contact', name: Contact},
    ]
---

export function Buttons({pages}) {
    const buttons = [];
    const btn_id = "nav-button-"
    
    for (let i in pages) {

        let page = pages[i]        
        let _btn_id = btn_id + i.toString()

        buttons.push(
            <div class='button-container'>
                <button class='nav-button'  id={_btn_id}>
                    <a class='nav-text' href={page.link}>
                        {page.name}
                    </a>
                </button>
            </div>
        );
            
    }
    
    return <>{buttons}</>;
}

<nav>
    <Buttons pages={frontmatter.pages} />
</nav>
